```c++
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;
int p;
int main()
{
    string str;
    vector<int> n;
    cin>>str;
    int i = 0;
    for(i = 97;i < 123;i++)
    {
        char a = i;
        n.push_back(count(str.begin(),str.end(),a));
    }
    auto position = max_element(n.begin(),n.end());
    for (i = 0; i < 26; i++)
    {
        if (n[i] == *position)
        {
            p = i;
            break;
        }
    }
    char max_char = p + 97;

    printf("%c\n",max_char);
    printf("%d\n",*position);
    return 0;
}
```
